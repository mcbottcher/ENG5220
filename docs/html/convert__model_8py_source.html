35 def transform_kernels(kernels, n_gates, transform_func):
36  """
37  Transforms CuDNN kernel matrices (either LSTM or GRU) into the regular Keras format.
38 
39  Parameters
40  ----------
41  kernels : numpy.ndarray
42  Composite matrix of input or recurrent kernels.
43  n_gates : int
44  Number of recurrent unit gates, 3 for GRU, 4 for LSTM.
45  transform_func: function(numpy.ndarray)
46  Function to apply to each input or recurrent kernel.
47 
48  Returns
49  -------
50  numpy.ndarray
51  Transformed composite matrix of input or recurrent kernels in C-contiguous layout.
52  """
53  return np.require(np.hstack([transform_func(kernel) for kernel in np.hsplit(kernels, n_gates)]), requirements='C')

701  """Check if backend used an offset while placing the filter
